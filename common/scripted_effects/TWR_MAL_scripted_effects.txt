MAL_add_malay_influence = {
	custom_effect_tooltip = MAL_add_malay_influence_tt
	if = {
		limit = {
			check_variable = { malay_influence < 100 }
		}
		add_to_variable = { malay_influence = temp_influence }
		subtract_from_variable = { chinese_influence = temp_influence }
		set_temp_variable = { temp_1 = temp_influence }
		multiply_temp_variable = { temp_1 = 0.005 }
		add_to_variable = { MAL_racial_tensions_conscription_factor = temp_1 }
		add_to_variable = { MAL_racial_tensions_production_speed_buildings_factor = temp_1 }
		subtract_from_variable = { MAL_racial_tensions_production_factory_max_efficiency_factor = temp_1 }
		set_temp_variable = { temp_1 = temp_influence }
		multiply_temp_variable = { temp_1 = 0.004 }
		add_to_variable = { MAL_racial_tensions_consumer_goods_factor = temp_1 }
		set_temp_variable = { temp_1 = temp_influence }
		multiply_temp_variable = { temp_1 = 0.01 }
		add_to_variable = { MAL_racial_tensions_political_power_gain = temp_1 }
		set_temp_variable = { temp_1 = temp_influence }
		multiply_temp_variable = { temp_1 = 0.002 }
		add_to_variable = { MAL_racial_tensions_research_speed_factor = temp_1 }
	}
}
