		country = USA
		relation = guarantee
		active = no
	}
	diplomatic_relation = {
		country = USA
		relation = military_access
		active = no
	}
	if = {
		limit = {
			is_in_faction_with = USA
		}
		leave_faction = yes
	}
	add_opinion_modifier = {
		target = USA
		modifier = relations_broken
	}
	reverse_add_opinion_modifier = {
		target = USA
		modifier = relations_broken
	}
	set_country_flag = relations_broken_with_usa
	remove_opinion_modifier = {
		target = USA
		modifier = russian_american_relations
	}
	USA = {
		remove_opinion_modifier = {
			target = RUS
			modifier = russian_american_relations
		}
	}
}

RUS_duma_bill_setup = {
	set_variable = { RUS_duma_interaction_progress = 100 }

	5 = {
		set_variable = { RUS_duma_support_percent = 0.50 }
	}

	if = {
		limit = {
			check_variable = { RUS_duma_amendment_count > 0 }
		}
		for_loop_effect = {
			start = 1
			end = RUS_duma_amendment_count
			compare = less_than_or_equals

			add_to_array = { RUS_duma_amendment_array = v }
		}
	}

	for_each_scope_loop = {
		array = RUS_duma_party_array

		set_variable = { RUS_duma_support_seat = RUS_duma_support_percent }
		multiply_variable = { RUS_duma_support_seat = RUS_duma_seat }
		round_variable = RUS_duma_support_seat
	}

	set_country_flag = RUS_duma_support_view

	RUS_duma_reload_support = yes

	hidden_effect = {
		# fire the timer
		country_event = { id = rus_duma.15 hours = 24 }
	}
}

RUS_duma_modify_support = {
	set_temp_variable = { temp_before = temp_party:RUS_duma_support_percent }
	set_temp_variable = { temp_after = temp_party:RUS_duma_support_percent }
	add_to_temp_variable = { temp_after = temp_support }

	if = {
		limit = {
			check_variable = { temp_after < 0 }
		}
		set_temp_variable = { temp_after = 0 }
	}
	else_if = {
		limit = {
			check_variable = { temp_after > 1 }
		}
		set_temp_variable = { temp_after = 1 }
	}

	set_variable = { temp_party:RUS_duma_support_percent = temp_after }

	set_variable = { temp_party:RUS_duma_support_seat = temp_party:RUS_duma_support_percent }
	multiply_variable = { temp_party:RUS_duma_support_seat = temp_party:RUS_duma_seat }
	round_variable = temp_party:RUS_duma_support_seat

	custom_effect_tooltip = RUS_duma_modify_support_tt

	RUS_duma_reload_support = yes
}

}

RUS_duma_add_seat = {
	custom_effect_tooltip = RUS_duma_add_seat_tt

	add_to_variable = { RUS_duma_seat_total = temp_seat_change }
	set_variable = { RUS_duma_seat_majority = RUS_duma_seat_total }
	divide_variable = { RUS_duma_seat_majority = 2 }
	round_variable = RUS_duma_seat_majority

	if = {
		limit = {
			set_temp_variable = { temp_adjustment = RUS_duma_seat_majority }
			multiply_temp_variable = { temp_adjustment = 2 }
			check_variable = { temp_adjustment < RUS_duma_seat_total }
		}
		add_to_variable = { RUS_duma_seat_majority = 1 }
	}

	6 = {
		set_variable = { RUS_duma_seat = 410 }
		subtract_from_variable = { RUS_duma_seat = ROOT.RUS_duma_seat_total }
	}

	for_each_loop = {
		array = RUS_duma_popularity_array

		set_temp_variable = { temp_popularity = RUS_duma_popularity_array^i }
		multiply_temp_variable = { temp_popularity = RUS_duma_seat_total }
		round_temp_variable = temp_popularity
		add_to_temp_variable = { temp_total = temp_popularity }
		add_to_temp_array = { temp_popularity_array = temp_popularity }
	}

	if = {
		limit = {
			check_variable = { temp_total < RUS_duma_seat_total }
		}
		set_temp_variable = { temp_seat = RUS_duma_seat_total }
		subtract_from_temp_variable = { temp_seat = temp_total }
		find_highest_in_array = { array = temp_popularity_array index = highest_index }
		add_to_temp_variable = { temp_popularity_array^highest_index = temp_seat }
	}
	else_if = {
		limit = {
			check_variable = { temp_total > RUS_duma_seat_total }
		}
		subtract_from_temp_variable = { temp_total = RUS_duma_seat_total }
		find_highest_in_array = { array = temp_popularity_array index = highest_index }
		subtract_from_temp_variable = { temp_popularity_array^highest_index = temp_total }
	}

	for_each_loop = {
		array = RUS_duma_party_array

		set_variable = { v:RUS_duma_seat = temp_popularity_array^i }
	}

	RUS_duma_reload_normal = yes
}

		multiply_temp_variable = { temp_popularity = RUS_duma_seat_total }
		round_temp_variable = temp_popularity
		add_to_temp_variable = { temp_total = temp_popularity }
		add_to_temp_array = { temp_popularity_array = temp_popularity }
	}

	if = {
		limit = {
			check_variable = { temp_total < RUS_duma_seat_total }
		}
		set_temp_variable = { temp_seat = RUS_duma_seat_total }
		subtract_from_temp_variable = { temp_seat = temp_total }
		find_highest_in_array = { array = temp_popularity_array index = highest_index }
		add_to_temp_variable = { temp_popularity_array^highest_index = temp_seat }
	}
	else_if = {
		limit = {
			check_variable = { temp_total > RUS_duma_seat_total }
		}
		subtract_from_temp_variable = { temp_total = RUS_duma_seat_total }
		find_highest_in_array = { array = temp_popularity_array index = highest_index }
		subtract_from_temp_variable = { temp_popularity_array^highest_index = temp_total }

		array = RUS_duma_party_array

		set_variable = { v:RUS_duma_seat = temp_popularity_array^i }

		if = {
			limit = {

					check_variable = { highest_index_later = 0 }
					check_variable = { highest_index_later = 1 }
				}
				OR = {
					check_variable = { i = 0 }
					check_variable = { i = 1 }
				}
			}
			set_variable = { v:RUS_duma_alignment = 1 }
			set_variable = { v:RUS_duma_alignment = 1 }
		}
		else_if = {
			limit = {
				check_variable = { i = highest_index_later }
			}
			set_variable = { v:RUS_duma_alignment = 1 }
		}
		else_if = {
			limit = {
				NOT = {
					check_variable = { v = 5 }
				}
			}
			set_variable = { v:RUS_duma_alignment = 4 }
		}
	}

	RUS_duma_reload_normal = yes
