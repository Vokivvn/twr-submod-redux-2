#Add to scripted triggers if more levels of hyper inflation is added
SPA_end_of_economics = {
	custom_effect_tooltip = SPA_economic_problems_ended
hidden_effect = {
	remove_ideas = {
		SPA_economic_stagnation
		SPA_economic_stagnation_1
		SPA_economic_stagnation_2
		SPA_economic_stagnation_3
	}
}
}

#Important to put HIGHEST first
SPA_economic_stagnation_improved = {
	custom_effect_tooltip = SPA_stagnation_up_tt
	if = {
		limit = { has_idea = SPA_economic_stagnation }
		swap_ideas = {
			remove_idea = SPA_economic_stagnation
			add_idea = SPA_economic_stagnation_1
		}
	}
	else_if = {
		limit = { has_idea = SPA_economic_stagnation_1 }
		swap_ideas = {
			remove_idea = SPA_economic_stagnation_1
			add_idea = SPA_economic_stagnation_2
		}
	}
	else_if = {
		limit = { has_idea = SPA_economic_stagnation_2 }
		swap_ideas = {
			remove_idea = SPA_economic_stagnation_2
			add_idea = SPA_economic_stagnation_3
		}
	}
	else_if = {
		limit = { has_idea = SPA_economic_stagnation_3 }
		remove_ideas = SPA_economic_stagnation_3	
	}

}
