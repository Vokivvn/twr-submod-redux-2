				for_loop_effect = {
					start = 1
					end = 4
					compare = less_than_or_equals

							NOT = {

			if = {
				limit = {
					check_variable = { 1.NRI_region_control_array^num > 1 }
					NOT = {
					check_variable = { 1.NRI_region_control_array^num < 2.NRI_region_control_array^num }
					check_variable = { 1.NRI_region_control_array^num < 3.NRI_region_control_array^num }
					check_variable = { 1.NRI_region_control_array^num < 4.NRI_region_control_array^num }
					}
				}
				custom_effect_tooltip = NRI_norilsk_campaign_LSRP_victory_tt
				country_event = twrnripol.1
			}
			else_if = {
				limit = {
					check_variable = { 2.NRI_region_control_array^num > 1 }
					NOT = {
					check_variable = { 2.NRI_region_control_array^num < 3.NRI_region_control_array^num }
					check_variable = { 2.NRI_region_control_array^num < 4.NRI_region_control_array^num }
					}

				custom_effect_tooltip = NRI_norilsk_campaign_AP_victory_tt
				country_event = twrnripol.2
			}
			else_if = {
				limit = {
					check_variable = { 3.NRI_region_control_array^num > 1 }
					NOT = {
					check_variable = { 3.NRI_region_control_array^num < 4.NRI_region_control_array^num }
					}
				}
				custom_effect_tooltip = NRI_norilsk_campaign_alternativa_victory_tt

			}
			else_if = {
				limit = {
					check_variable = { 4.NRI_region_control_array^num > 1 }
				}

			}
			else = {
				custom_effect_tooltip = NRI_norilsk_campaign_nobody_wins_tt
