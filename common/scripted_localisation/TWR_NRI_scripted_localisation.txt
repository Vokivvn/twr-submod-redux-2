defined_text = {
	name = GetNRIFactionList
	text = {
		trigger = {
			set_temp_variable = { cur = -1 }
			set_temp_variable = { global.temp_region_id = THIS }
		}
		localization_key = "[GetNRIFactionEntry]"
	}
}

defined_text = {
	name = GetNRIFactionEntry
	text = {
		trigger = {
			add_to_temp_variable = { cur = 1 }
			check_variable = { global.temp_region_id:NRI_norilsk_campaign_faction_array^num > cur }
			set_temp_variable = { temp_faction = global.temp_region_id:NRI_norilsk_campaign_faction_array^cur }
			set_temp_variable = { temp_control = temp_faction:NRI_faction_control@global.temp_region_id }
			set_temp_variable = { temp_resource = temp_faction:NRI_faction_resource@global.temp_region_id }
		}
        localization_key = NRI_norilsk_campaign_faction_entry
    }
	text = {
		localization_key = ""
	}
}
