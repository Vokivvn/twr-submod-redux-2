division_limiter = {
	enable = {
		set_temp_variable = {
			upper_limit = num_of_factories
		}
		multiply_temp_variable = {
			upper_limit = 1.15
		}
		if = {
			limit = {
				has_war = yes
			}
			multiply_temp_variable = {
				upper_limit = 2
			}
		}
		if = {
			limit = {
				is_major = yes
			}
			multiply_temp_variable = {
				upper_limit = 1.25
			}
		}
		if = {
			limit = {
				threat > 0.75
			}
			multiply_temp_variable = {
				upper_limit = 1.15
			}
		}
		check_variable = {
			num_divisions > upper_limit
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = build_army
		id = garrison
		value = -1000
	}

	ai_strategy = {
		type = build_army
		id = suppression
		value = -1000
	}

	ai_strategy = {
		type = build_army
		id = paratroopers
		value = -1000
	}
	
	ai_strategy = {
		type = build_army
		id = mountaineers
		value = -1000
	}

		value = -1000
	}
