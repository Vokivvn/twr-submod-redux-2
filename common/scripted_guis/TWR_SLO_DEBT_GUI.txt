scripted_gui = {

	TWR_SLO_DEBT_Scripted_GUI = {
		context_type = decision_category
		window_name = "TWR_SLO_DEBT_GUI"

		properties = {
			parliament_icon = {
				frame = current_house_icon_frame
			}
		}

                triggers = {
			SLO_DEBT_HIGH_visible = {
                               OR = {
                               AND = {
                               check_variable = { SLO_Koruna < 45 }
                               check_variable = { SLO_Koruna > 35 }
                               }
                               check_variable = { SLO_Koruna = 35 } 
                               check_variable = { SLO_Koruna = 45 }
                               }
			}
		}
                triggers = {
			SLO_DEBT_MEDIUM_visible = {
                               OR = {
                               AND = {
                               check_variable = { SLO_Koruna < 35 }
                               check_variable = { SLO_Koruna > 20 }
                               }
                               check_variable = { SLO_Koruna = 20 }
                               }
			}
		}
                triggers = {
			SLO_DEBT_LOW_visible = {
                               OR = {
                               AND = {
                               check_variable = { SLO_Koruna < 20 }
                               check_variable = { SLO_Koruna > 1 }
                               }
                               check_variable = { SLO_Koruna = 20 }
                               check_variable = { SLO_Koruna = 1 }
                               }
			}
		}
                triggers = {
			SLO_DEBT_EMPTY_visible = {
                                check_variable = { SLO_Koruna = 0 }
			}
		}
	}

        TWR_SLO_ARMY = {
		context_type = decision_category
		window_name = "TWR_SLO_ARMY"

		properties = {
			parliament_icon = {
				frame = current_house_icon_frame
			}
		}
        }

}
