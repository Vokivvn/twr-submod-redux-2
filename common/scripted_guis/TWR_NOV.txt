scripted_gui = { #Boofchev Gui
	NOV_ps_open_button = {
		context_type = player_context
		parent_window_token = politics_tab
		window_name = "NOV_ps_open_button_window"

		visible = {
			tag = NOV
			has_country_flag = NOV_MECHANIC_ENABLE
			NOT = {
				has_country_flag = NOV_MECHANIC_DISABLE
			}
		}

		effects = {
			politicsview_NOV_button_click = {
				if = {
					limit = { has_country_flag = NOV_Khrushchev_mechanic_open }
					clr_country_flag = NOV_Khrushchev_mechanic_open
				}
				else = {
					set_country_flag = NOV_Khrushchev_mechanic_open
				}
			}
		}
	}
	
	NOV_power_struggle_gui = {
		context_type = player_context
		parent_window_token = politics_tab
		window_name = "NOV_ps_screen_window"

		visible = {
			has_country_flag = NOV_Khrushchev_mechanic_open
			NOT = {
				has_country_flag = NOV_MECHANIC_DISABLE
			}
		}

		effects = {
			NOV_close_ps_button_click = {
				clr_country_flag = NOV_Khrushchev_mechanic_open
			}
		}
		
		properties = {
			NOV_idealist_leader = {
				image = "[NOV_get_idealist_leader]"
			}
			NOV_realist_leader = {
				image = "[NOV_get_realist_leader]"
			}
		}
	}
}
