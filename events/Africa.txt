	id = africa.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event africa.1"}

	desc = africa.1.d
	picture = GFX_report_event_barotseland
	
	fire_only_once = yes
	
	mean_time_to_happen = { days = 30 }

	trigger = {
		tag = ZAM
		ZAM = { owns_state = 959 }
	}
	
	option = { #They deserve their rights
		name = africa.1.a
		ai_chance = {
			factor = 50
		}
		add_ideas = ZAM_Lozi_Autonomous_Region
	}
	
	option = { #Dont give them the rights
		name = africa.1.b
		ai_chance = {
			factor = 50
		}
		random_list = {
			75 = {

					id = africa.2
					days = 1
					random = 24
				}
			}
			25 = {
				country_event = {
					id = africa.3
					days = 1
					random = 24
				}
			}
		}
	}
}
